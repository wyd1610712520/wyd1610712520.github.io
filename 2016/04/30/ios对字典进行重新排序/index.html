ios对字典进行重新排序

发表于 2016-04-30   |   分类于 ios   |   
 NSArray *keys = [self.selectHoleIdAndResultDict allKeys];               
NSArray *sortedArray = [keys sortedArrayUsingComparator:^NSComparisonResult(id obj1, id obj2) {
NSString *firstStr = [NSString stringWithFormat:@"%@",obj1];
NSString *secondStr = [NSString stringWithFormat:@"%@",obj2];
    int first = [firstStr intValue];
    int second = [secondStr intValue];
    NSLog(@">>>>>%@--%@>>>>>>>>>%d------%d",obj1,obj2,first,second);
    return [@(first) compare:@(second)];
}];
for (NSString *categoryId in sortedArray) {
